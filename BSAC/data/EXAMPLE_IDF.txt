
for p in $HPATH/LOG $DPATH/FIT $DPATH/SYNTHESIS $DPATH/ANALYSIS $DPATH/CONSTRAIN $DPATH/FIGURES
do
    if [[ ! -d $p ]]
    then
	    mkdir $p
	fi
done

## Find the model list
MODELS=$(ls $DPATH/INPUT/IDF/X-GMST)

## Loop on models to fit the statistical model
echo "Loop on models"
for MODNC in $MODELS
do
    MOD=$(basename $MODNC .nc)
    echo " * $MOD"
    echo "   => fit X"
    bsac fit X --n-samples $N_SAMPLES --log info $HPATH/LOG/'FITX_'$MOD.log\
               --bias-period $BIAS_PERIOD\
               --input GMST,$DPATH/INPUT/IDF/X-GMST/$MODNC EU,$DPATH/INPUT/IDF/X-EU/$MODNC\
               --save-clim $DPATH/FIT/$MOD'_fitX.nc'\
               --common-period historical --different-periods ssp126,ssp245,ssp370,ssp585\
               --config GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE\
               --total-memory $TOTAL_MEMORY\
               --n-workers $N_WORKERS\
               --threads-per-worker $THREADS_PER_WORKER
	
    echo "   => show X"
    bsac show X --n-samples $N_SAMPLES --log info $HPATH/LOG/'SHOWX_'$MOD.log\
                --bias-period $BIAS_PERIOD\
                --load-clim $DPATH/FIT/$MOD'_fitX.nc'\
                --input GMST,$DPATH/INPUT/IDF/X-GMST/$MODNC EU,$DPATH/INPUT/IDF/X-EU/$MODNC\
                --output $DPATH/FIGURES/BSAC_SHOW_X_$MOD.pdf\
                --total-memory $TOTAL_MEMORY\
                --n-workers $N_WORKERS\
                --threads-per-worker $THREADS_PER_WORKER
    
    echo "   => fit Y"
    bsac fit Y --n-samples $N_SAMPLES --log info $HPATH/LOG/'FITY_'$MOD.log\
               --bias-period $BIAS_PERIOD\
               --input $DPATH/INPUT/IDF/Y-IDF/$MODNC\
		       --config name=IDF,spatial=lat:lon,nslaw=GEV,GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE\
               --load-clim $DPATH/FIT/$MOD'_fitX.nc'\
               --save-clim $DPATH/FIT/$MOD'_fitY.nc'\
               --common-period historical --different-periods ssp126,ssp245,ssp370,ssp585\
               --total-memory $TOTAL_MEMORY\
               --n-workers $N_WORKERS\
               --threads-per-worker $THREADS_PER_WORKER

done

### Run synthesis
echo " * synthesis"
GRID=$DPATH/INPUT/IDF/infos/mask_IDF.nc
GRIDNAME='area_fraction'
bsac synthesize --log info $HPATH/LOG/SYNTHESIS.log\
                --bias-period $BIAS_PERIOD\
                --input $DPATH/FIT/*_fitY.nc\
                --common-period historical --different-periods ssp126,ssp245,ssp370,ssp585\
                --config grid=$GRID,grid_name=$GRIDNAME,spatial=lat:lon,nslaw=GEV,GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE,names=GMST:EU:IDF\
                --save-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
                --total-memory $TOTAL_MEMORY\
                --n-workers $N_WORKERS\
                --threads-per-worker $THREADS_PER_WORKER

echo " * show X of synthesis"
bsac show X --n-samples $N_SAMPLES --log info $HPATH/LOG/SHOWX_SYNTHESIS.log\
            --bias-period $BIAS_PERIOD\
            --load-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
            --output $DPATH/FIGURES/BSAC_SHOW_X_SYNTHESIS.pdf\
            --total-memory $TOTAL_MEMORY\
            --n-workers $N_WORKERS\
            --threads-per-worker $THREADS_PER_WORKER

## Run constraint
echo " * constraint X of observations"
bsac constrain X --log info $HPATH/LOG/CONSTRAINX.log\
                 --bias-period $BIAS_PERIOD\
                 --load-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
                 --save-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
                 --input GMST,$DPATH/INPUT/IDF/Xo/GMST_GISTEMP_tas_year_1880-2023.nc EU,$DPATH/INPUT/IDF/Xo/EU_HadCRUT_tas_year_1850-2023.nc\
                 --total-memory $TOTAL_MEMORY\
                 --n-workers $N_WORKERS\
                 --threads-per-worker $THREADS_PER_WORKER

echo " * show X of constraint"
bsac show X --n-samples $N_SAMPLES --log info $HPATH/LOG/SHOWX_CONSTRAINX.log\
            --bias-period $BIAS_PERIOD\
            --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
            --input $DPATH/INPUT/GMST/Xo/GISTEMP_tas_year_1880-2023.nc\
            --output $DPATH/FIGURES/BSAC_SHOW_X_CONSTRAINX.pdf\
            --total-memory $TOTAL_MEMORY\
            --n-workers $N_WORKERS\
            --threads-per-worker $THREADS_PER_WORKER

echo " * show X of constraint VS synthesis"
bsac show CX --n-samples $N_SAMPLES --log info $HPATH/LOG/SHOWCX_CONSTRAINX-VS-SYNTHESIS.log\
             --bias-period $BIAS_PERIOD\
             --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
             --input $DPATH/SYNTHESIS/SYNTHESIS.nc\
             --output $DPATH/FIGURES/BSAC_SHOW_CX_CONSTRAINX-VS-SYNTHESIS.pdf\
             --total-memory $TOTAL_MEMORY\
             --n-workers $N_WORKERS\
             --threads-per-worker $THREADS_PER_WORKER

echo " * constraint Y of observations"
bsac constrain Y --log info $HPATH/LOG/CONSTRAINY.log\
                 --n-samples $N_SAMPLES\
                 --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
                 --save-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
                 --input TX3x,$DPATH/INPUT/IDF/Yo/IDF_ERA5_TX3x_year_1940-2023.nc\
                 --output $DPATH/CONSTRAIN/SAMPLES_CONSTRAINY.nc\
                 --config size-chain=10\
                 --total-memory $TOTAL_MEMORY\
                 --n-workers $N_WORKERS\
                 --threads-per-worker $THREADS_PER_WORKER

## Attribution
bsac attribute freturnt --log info $HPATH/LOG/ANALYSIS_ATTR-FRETURNT.log\
                        --n-samples $N_SAMPLES\
                        --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
	                    --input 2,10,30,50,100,1000\
                        --output $DPATH/ANALYSIS/FRETURNT.nc\
                        --config mode=quantile,side=right\
                        --total-memory $TOTAL_MEMORY\
                        --n-workers $N_WORKERS\
                        --threads-per-worker $THREADS_PER_WORKER

bsac attribute fintensity --log info $HPATH/LOG/ANALYSIS_ATTR-FINTENSITY.log\
                          --n-samples $N_SAMPLES\
                          --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
	                      --input TX3x,$DPATH/INPUT/IDF/Yo/IDF_ERA5_TX3x_xyear_1940-2023.nc\
                          --output $DPATH/ANALYSIS/FINTENSITY-MAP.nc\
                          --config mode=quantile,side=right\
                          --total-memory $TOTAL_MEMORY\
                          --n-workers $N_WORKERS\
                          --threads-per-worker $THREADS_PER_WORKER

bsac attribute fintensity --log info $HPATH/LOG/ANALYSIS_ATTR-FINTENSITY.log\
                          --n-samples $N_SAMPLES\
                          --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
	                      --input 323.15\
                          --output $DPATH/ANALYSIS/FINTENSITY-VALUE.nc\
                          --config mode=quantile,side=right\
                          --total-memory $TOTAL_MEMORY\
                          --n-workers $N_WORKERS\
                          --threads-per-worker $THREADS_PER_WORKER

bsac attribute event --log info $HPATH/LOG/ANALYSIS_ATTR-EVENT2019.log\
                     --n-samples $N_SAMPLES\
                     --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
                     --input TX3x,$DPATH/INPUT/IDF/Yo/IDF_ERA5_TX3x_year_1940-2023.nc\
                     --output $DPATH/ANALYSIS/EVENT2019.nc\
                     --config time=2019,mode=quantile,side=right\
                     --total-memory $TOTAL_MEMORY\
                     --n-workers $N_WORKERS\
                     --threads-per-worker $THREADS_PER_WORKER

bsac draw --log info $HPATH/LOG/ANALYSIS_DRAW.log\
          --n-samples $N_SAMPLES\
          --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
          --output $DPATH/ANALYSIS/DRAW.nc\
          --total-memory $TOTAL_MEMORY\
          --n-workers $N_WORKERS\
          --threads-per-worker $THREADS_PER_WORKER

## END OF SCRIPT
