
## Fixed Parameters
BIAS_PERIOD='1900/1950'
GAM_DOF=7
GAM_DEGREE=3

## Clean and build path
for p in $HPATH/LOG $DPATH/FIT $DPATH/SYNTHESIS $DPATH/ANALYSIS $DPATH/CONSTRAIN $DPATH/FIGURES
do
    if [[ ! -d $p ]]
    then
	    mkdir $p
	fi
done

## Some parameters
DPERS='ssp126,ssp245,ssp370,ssp585'

## Find the model list
MODELS=$(ls $DPATH/INPUT/X-GMST)

## Loop on models to fit the statistical model
echo "Loop on models"
for MODNC in $MODELS
do
    MOD=$(basename $MODNC .nc)
    echo " * $MOD"
    echo "   => fit X"
    bsac fit X --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/'FITX_'$MOD.log\
               --bias-period $BIAS_PERIOD\
               --input GMST,$DPATH/INPUT/X-GMST/$MODNC EU,$DPATH/INPUT/X-EU/$MODNC\
               --save-clim $DPATH/FIT/$MOD'_fitX.nc'\
               --common-period historical --different-periods $DPERS\
               --config GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE\
               --total-memory $TOTAL_MEMORY\
               --n-workers $N_WORKERS\
               --threads-per-worker $THREADS_PER_WORKER
	
    echo "   => show X"
    bsac show X --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/'SHOWX_'$MOD.log\
                --bias-period $BIAS_PERIOD\
                --load-clim $DPATH/FIT/$MOD'_fitX.nc'\
                --input GMST,$DPATH/INPUT/X-GMST/$MODNC EU,$DPATH/INPUT/X-EU/$MODNC\
                --output $DPATH/FIGURES/BSAC_SHOW_X_$MOD.pdf\
                --total-memory $TOTAL_MEMORY\
                --n-workers $N_WORKERS\
                --threads-per-worker $THREADS_PER_WORKER
    
    echo "   => fit Y"
    bsac fit Y --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/'FITY_'$MOD.log\
               --bias-period $BIAS_PERIOD\
               --input $DPATH/INPUT/Y/$MODNC\
		       --config name=TX3x,nslaw=GEV,GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE\
               --load-clim $DPATH/FIT/$MOD'_fitX.nc'\
               --save-clim $DPATH/FIT/$MOD'_fitY.nc'\
               --common-period historical --different-periods $DPERS\
               --total-memory $TOTAL_MEMORY\
               --n-workers $N_WORKERS\
               --threads-per-worker $THREADS_PER_WORKER
	
done

## Run synthesis
echo " * synthesis"
bsac synthesize -v --log-file $HPATH/LOG/SYNTHESIS.log\
                --bias-period $BIAS_PERIOD\
                --input $DPATH/FIT/*_fitY.nc\
                --common-period historical --different-periods $DPERS\
                --config nslaw=GEV,GAM_dof=$GAM_DOF,GAM_degree=$GAM_DEGREE,names=GMST:EU:TX3x\
                --save-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
                --total-memory $TOTAL_MEMORY\
                --n-workers $N_WORKERS\
                --threads-per-worker $THREADS_PER_WORKER

echo " * show X of synthesis"
bsac show X --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/SHOWX_SYNTHESIS.log\
            --bias-period $BIAS_PERIOD\
            --load-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
            --output $DPATH/FIGURES/BSAC_SHOW_X_SYNTHESIS.pdf\
            --total-memory $TOTAL_MEMORY\
            --n-workers $N_WORKERS\
            --threads-per-worker $THREADS_PER_WORKER

## Run constraint
echo " * constraint X of observations"
bsac constrain X -v --log-file $HPATH/LOG/CONSTRAINX.log\
                 --bias-period $BIAS_PERIOD\
                 --load-clim $DPATH/SYNTHESIS/SYNTHESIS.nc\
                 --save-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
                 --input GMST,$DPATH/INPUT/Xo/GMST_GISTEMP_tas_year_1880-2023.nc EU,$DPATH/INPUT/Xo/EU_HadCRUT_tas_year_1850-2023.nc\
                 --total-memory $TOTAL_MEMORY\
                 --n-workers $N_WORKERS\
                 --threads-per-worker $THREADS_PER_WORKER

echo " * show X of constraint"
bsac show X --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/SHOWX_CONSTRAINX.log\
            --bias-period $BIAS_PERIOD\
            --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
            --input $DPATH/INPUT/GMST/Xo/GISTEMP_tas_year_1880-2023.nc\
            --output $DPATH/FIGURES/BSAC_SHOW_X_CONSTRAINX.pdf\
            --total-memory $TOTAL_MEMORY\
            --n-workers $N_WORKERS\
            --threads-per-worker $THREADS_PER_WORKER

echo " * show X of constraint VS synthesis"
bsac show CX --n-samples $N_SAMPLES -v --log-file $HPATH/LOG/SHOWCX_CONSTRAINX-VS-SYNTHESIS.log\
             --bias-period $BIAS_PERIOD\
             --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
             --input $DPATH/SYNTHESIS/SYNTHESIS.nc\
             --output $DPATH/FIGURES/BSAC_SHOW_CX_CONSTRAINX-VS-SYNTHESIS.pdf\
             --total-memory $TOTAL_MEMORY\
             --n-workers $N_WORKERS\
             --threads-per-worker $THREADS_PER_WORKER

echo " * constraint Y of observations"
bsac constrain Y -v --log-file $HPATH/LOG/CONSTRAINY.log\
                 --n-samples $N_SAMPLES\
                 --bias-period $BIAS_PERIOD\
                 --load-clim $DPATH/CONSTRAIN/CONSTRAINX.nc\
                 --save-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
                 --input TX3x,$DPATH/INPUT/Yo/MF-stations_Paris-Montsouris_TX3x_year_1872-2023.nc\
                 --output $DPATH/CONSTRAIN/SAMPLES_CONSTRAINY.nc\
                 --config size-chain=10\
                 --total-memory $TOTAL_MEMORY\
                 --n-workers $N_WORKERS\
                 --threads-per-worker $THREADS_PER_WORKER

## Draw parameters
echo " * Draw"
bsac draw -v --log-file $HPATH/LOG/ANALYSIS_DRAW.log\
          --n-samples $N_SAMPLES\
          --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
          --output $DPATH/ANALYSIS/DRAW.nc\
          --total-memory $TOTAL_MEMORY\
          --n-workers $N_WORKERS\
          --threads-per-worker $THREADS_PER_WORKER

## Attribution
echo " * Attribution: freturnt"
bsac attribute freturnt -v --log-file $HPATH/LOG/ANALYSIS_ATTR-FRETURNT.log\
                        --n-samples $N_SAMPLES\
                        --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
	                    --input 2,10,30,50,100,1000\
                        --output $DPATH/ANALYSIS/FRETURNT.nc\
                        --config mode=quantile,side=right\
                        --total-memory $TOTAL_MEMORY\
                        --n-workers $N_WORKERS\
                        --threads-per-worker $THREADS_PER_WORKER


echo " * Attribution: fintensity"
bsac attribute fintensity -v --log-file $HPATH/LOG/ANALYSIS_ATTR-FINTENSITY.log\
                          --n-samples $N_SAMPLES\
                          --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
	                      --input 323.15\
                          --output $DPATH/ANALYSIS/FINTENSITY-VALUE.nc\
                          --config mode=quantile,side=right\
                          --total-memory $TOTAL_MEMORY\
                          --n-workers $N_WORKERS\
                          --threads-per-worker $THREADS_PER_WORKER

echo " * Attribution: event"
bsac attribute event -v --log-file $HPATH/LOG/ANALYSIS_ATTR-EVENT2019.log\
                     --n-samples $N_SAMPLES\
                     --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
                     --input TX3x,$DPATH/INPUT/Yo/MF-stations_Paris-Montsouris_TX3x_year_1872-2023.nc\
                     --output $DPATH/ANALYSIS/EVENT2019.nc\
                     --config time=2019,mode=quantile,side=right\
                     --total-memory $TOTAL_MEMORY\
                     --n-workers $N_WORKERS\
                     --threads-per-worker $THREADS_PER_WORKER

echo " * WPE"
bsac misc wpe -v --log-file $HPATH/LOG/WPE.log\
              --n-samples $N_SAMPLES\
              --load-clim $DPATH/CONSTRAIN/CONSTRAINY.nc\
              --input IPCC\
              --output $DPATH/ANALYSIS/WPE.nc\
              --config period=2024/2060,mode=quantile,ci=0.05\
              --total-memory $TOTAL_MEMORY\
              --n-workers $N_WORKERS\
              --threads-per-worker $THREADS_PER_WORKER


## END OF SCRIPT
